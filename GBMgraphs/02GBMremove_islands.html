def removeIslands(matrix):
  rows,cols=len(matrix),len(matrix[0])
  safe=set() # border-connected 1s
  def floodFill(r,c):
   if (r,c)in safe: return
   if matrix[r][c]==0: return
   safe.add((r,c))
   for dr,dc in [(-1,0),(1,0),(0,-1),(0,1)]:
    nr,nc=r+dr,c+dc
    if 0<=nr<rows and 0<=nc<cols:
     floodFill(nr,nc)
  # Step 1: flood all border 1s
  for r in range(rows):
   for c in range(cols):
    if rin[0,rows-1]or cin[0,cols-1]:
     if matrix[r][c]==1: floodFill(r,c)
  # Step 2: remove non-safe 1s
  for r in range(rows):
   for c in range(cols):
    if matrix[r][c]==1 and (r,c)not in safe:
     matrix[r][c]=0  # island → remove
  return matrix

#pragma GCC optimize("O3")
#include<bits/stdc++.h>
using namespace std;

vector<vector<int>> removeIslands(
    vector<vector<int>>& m){
  int R=m.size(), C=m[0].size();
  vector<vector<bool>> safe(R,vector<bool>(C,false));
  // DFS flood fill from border
  auto dfs=[&](auto& self,int r,int c)->void{
    if(r<0||r>=R||c<0||c>=C) return;
    if(safe[r][c]||m[r][c]==0) return;
    safe[r][c]=true;
    for(auto[dr,dc]:{make_pair(-1,0),{1,0},{0,-1},{0,1}})
      self(self,r+dr,c+dc);
  };
  // Flood from all border cells
  for(int r=0;r<R;r++)
    for(int c=0;c<C;c++)
      if((r==0||r==R-1||c==0||c==C-1)&&m[r][c]==1)
        dfs(dfs,r,c);
  // Remove islands (non-safe 1s)
  for(int r=0;r<R;r++)
    for(int c=0;c<C;c++)
      if(m[r][c]==1&&!safe[r][c])
        m[r][c]=0;
  return m;
}
