JavaScript — Single Cycle Check
function singleCycleCheck(array) {
  const n = array.length;
  let numVisited = 0;
  let currentIdx = 0;
  while (numVisited < n) {
   if (numVisited > 0 && currentIdx === 0)
    return false; // returned early
   numVisited++;
   currentIdx = getNextIdx(currentIdx);
  }
  return currentIdx === 0;
}
function getNextIdx(idx) {
  const jump = array[idx];
  const nextIdx = (idx + jump) % n;
  return nextIdx >= 0
   ? nextIdx : nextIdx + n;
}

#pragma GCC optimize("O3")
#include<bits/stdc++.h>
using namespace std;

bool singleCycleCheck(vector<int>& a){
  int n=a.size(), visited=0, idx=0;
  auto nxt=[&](int i){
    int j=(i+a[i])%n;
    return j<0?j+n:j;  // handle negative mod
  };
  while(visited<n){
    if(visited>0&&idx==0) return false; // early return to 0
    visited++;
    idx=nxt(idx);
  }
  return idx==0;  // must land exactly on 0
}

int main(){
  ios::sync_with_stdio(false); cin.tie(nullptr);
  int n; cin>>n;
  vector<int> a(n);
  for(auto& x:a) cin>>x;
  cout<<(singleCycleCheck(a)?"true":"false")<<'\n';
}
