def riverSizes(matrix):
  R,C=len(matrix),len(matrix[0])
  visited=[[False]*C for _ in range(R)]
  sizes=[]
  for r in range(R):
   for c in range(C):
    if visited[r][c]: continue
    if matrix[r][c]==0:
     visited[r][c]=True; continue
    # DFS flood fill this river
    stack=[(r,c)]; size=0
    while stack:
     cr,cc=stack.pop()
     if visited[cr][cc]: continue
     visited[cr][cc]=True
     if matrix[cr][cc]==1: size+=1
     for dr,dc in [(-1,0),(1,0),(0,-1),(0,1)]:
      nr,nc=cr+dr,cc+dc
      if 0<=nr<R and 0<=nc<C \
         and not visited[nr][nc]:
       stack.append((nr,nc))
    if size>0: sizes.append(size)
  return sizes

#pragma GCC optimize("O3")
#include<bits/stdc++.h>
using namespace std;

vector<int> riverSizes(
    vector<vector<int>>& m){
  int R=m.size(),C=m[0].size();
  vector<vector<bool>> vis(R,vector<bool>(C,false));
  int dx[]={-1,1,0,0}, dy[]={0,0,-1,1};
  vector<int> sizes;
  for(int r=0;r<R;r++){
    for(int c=0;c<C;c++){
      if(vis[r][c]||m[r][c]==0){vis[r][c]=true;continue;}
      // DFS iterative
      stack<pair<int,int>> stk;
      stk.push({r,c}); int sz=0;
      while(!stk.empty()){
        auto [cr,cc]=stk.top(); stk.pop();
        if(vis[cr][cc]) continue;
        vis[cr][cc]=true;
        if(m[cr][cc]==1) sz++;
        for(int d=0;d<4;d++){
          int nr=cr+dx[d],nc=cc+dy[d];
          if(nr>=0&&nr<R&&nc>=0&&nc<C&&!vis[nr][nc])
            stk.push({nr,nc});
        }
      }
      if(sz) sizes.push_back(sz);
    }
  }
  return sizes;
}
