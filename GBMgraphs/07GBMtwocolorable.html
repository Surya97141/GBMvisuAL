Python — Two-Colorable (Bipartite BFS)
def twoColorable(edges):
  n=len(edges)
  color=[-1]*n # -1 = uncolored
  for start in range(n): # handle disconnected
   if color[start]!=-1: continue
   color[start]=0
   queue=deque([start])
   while queue:
    node=queue.popleft()
    for nb in edges[node]:
     if nb==node: return False # self-loop
     if color[nb]==color[node]:
      return False # conflict!
     if color[nb]==-1:
      color[nb]=1-color[node]
      queue.append(nb)
  return True

#pragma GCC optimize("O3")
#include<bits/stdc++.h>
using namespace std;

bool twoColorable(
    vector<vector<int>>& adj){
  int n=adj.size();
  vector<int> color(n,-1);
  for(int s=0;s<n;s++){
    if(color[s]!=-1) continue;
    color[s]=0;
    queue<int> q;
    q.push(s);
    while(!q.empty()){
      int u=q.front(); q.pop();
      for(int v:adj[u]){
        if(v==u) return false;       // self-loop
        if(color[v]==color[u]) return false; // conflict!
        if(color[v]==-1){
          color[v]=1-color[u];        // flip color
          q.push(v);
        }
      }
    }
  }
  return true;
}
