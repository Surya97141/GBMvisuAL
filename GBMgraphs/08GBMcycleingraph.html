def cycleInGraph(edges):
  n=len(edges)
  WHITE,GREY,BLACK=0,1,2
  color=[WHITE]*n
  def dfs(node):
   color[node]=GREY  # enter DFS stack
   for nb in edges[node]:
    if color[nb]==GREY:
     return True  # back-edge → cycle!
    if color[nb]==WHITE:
     if dfs(nb): return True
   color[node]=BLACK  # all paths explored
   return False
  for node in range(n):
   if color[node]==WHITE:
    if dfs(node): return True
  return False

#pragma GCC optimize("O3")
#include<bits/stdc++.h>
using namespace std;

enum Color { WHITE, GREY, BLACK };

bool dfs(vector<vector<int>>& adj,
         vector<Color>& col, int u){
  col[u]=GREY;           // enter recursion stack
  for(int v:adj[u]){
    if(col[v]==GREY) return true;  // back-edge
    if(col[v]==WHITE)
      if(dfs(adj,col,v)) return true;
  }
  col[u]=BLACK;           // leave recursion stack
  return false;
}

bool cycleInGraph(
    vector<vector<int>>& adj){
  int n=adj.size();
  vector<Color> col(n,WHITE);
  for(int i=0;i<n;i++)
    if(col[i]==WHITE)
      if(dfs(adj,col,i)) return true;
  return false;
}
