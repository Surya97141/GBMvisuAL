def minimumPassesOfMatrix(matrix):
  R,C=len(matrix),len(matrix[0])
  # seed queue with ALL initial positives
  queue=deque()
  for r in range(R):
   for c in range(C):
    if matrix[r][c]>0: queue.append((r,c))
  passes=0
  while queue:
   sz=len(queue) # snapshot current level
   hasConverted=False
   for _ in range(sz):
    r,c=queue.popleft()
    for dr,dc in [(-1,0),(1,0),(0,-1),(0,1)]:
     nr,nc=r+dr,c+dc
     if 0<=nr<R and 0<=nc<C \
        and matrix[nr][nc]<0:
      matrix[nr][nc]*=-1  # flip!
      queue.append((nr,nc))
      hasConverted=True
   if hasConverted: passes+=1
  negLeft=any(matrix[r][c]<0
   for r in range(R)
   for c in range(C))
  return -1 if negLeft else passes

#include<bits/stdc++.h>

int minPasses(vector<vector<int>>& m){
  int R=m.size(),C=m[0].size();
  queue<pair<int,int>> q;
  for(int r=0;r<R;r++)
    for(int c=0;c<C;c++)
      if(m[r][c]>0) q.push({r,c});
  int passes=0;
  int dx[]={-1,1,0,0},dy[]={0,0,-1,1};
  while(!q.empty()){
    int sz=q.size();
    bool converted=false;
    while(sz--){
      auto [r,c]=q.front(); q.pop();
      for(int d=0;d<4;d++){
        int nr=r+dx[d],nc=c+dy[d];
        if(nr<0||nr>=R||nc<0||nc>=C) continue;
        if(m[nr][nc]<0){
          m[nr][nc]*=-1;          // flip
          q.push({nr,nc});
          converted=true;
        }
      }
    }
    if(converted) passes++;
  }
  for(auto& row:m)
    for(int v:row)
      if(v<0) return -1;
  return passes;
}
