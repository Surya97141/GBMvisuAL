C++ — Rectangle Mania
#pragma GCC optimize("O3")
#include<bits/stdc++.h>
using namespace std;
int rectangleMania(
    vector<pair<int,int>>& coords){
  set<pair<int,int>> pts(
    coords.begin(),coords.end());
  int count=0;
  for(int i=0;i<coords.size();i++)
   for(int j=i+1;j<coords.size();j++){
    auto [x1,y1]=coords[i]; // p1
    auto [x2,y2]=coords[j]; // p2
    if(x1==x2 || y1==y2) continue;
    int bx=min(x1,x2),by=min(y1,y2);
    int tx=max(x1,x2),ty=max(y1,y2);
    if(pts.count({bx,ty}) &&
       pts.count({tx,by}))
     count++;
   }
  return count/2; // each rect counted twice
}
⚡ Why divide by 2?
// Each rectangle is counted TWICE:
// Once when (BL,TR) is the pair
// Once when (TR,BL) is the pair
// Since j > i, each pair appears once,
// but both diagonals of a rectangle
// are tested — so divide by 2.

// unordered_set version (O(1) avg):

struct PHash {
  size_t operator()(pair<int,int> p)const{
    return hash<long long>()(((long long)p.first<<32)|p.second);
  }
};
using PSet=unordered_set<pair<int,int>,PHash>;

int rectangleManiaFast(
    vector<pair<int,int>>& coords){
  PSet pts(coords.begin(),coords.end());
  int cnt=0;
  for(int i=0;i<(int)coords.size();i++)
    for(int j=i+1;j<(int)coords.size();j++){
      auto [x1,y1]=coords[i];
      auto [x2,y2]=coords[j];
      if(x1==x2||y1==y2) continue;
      int bx=min(x1,x2),by=min(y1,y2);
      int tx=max(x1,x2),ty=max(y1,y2);
      if(pts.count({bx,ty})&&pts.count({tx,by}))
        cnt++;
    }
  return cnt/2;
}
