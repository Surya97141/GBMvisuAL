class TrieNode:
  children={}; word=None
def insert(root,word):
  node=root
  for ch in word:
   if ch not in node.children:
    node.children[ch]=TrieNode()
   node=node.children[ch]
  node.word=word

struct Trie {
  Trie* ch[128]{}; string word;
};
void ins(Trie* r, const string& w){
  for(char c:w){
    if(!r->ch[c]) r->ch[c]=new Trie;
    r=r->ch[c];
  }
  r->word=w;
}
int dx[]={-1,-1,-1,0,0,1,1,1};
int dy[]={-1,0,1,-1,1,-1,0,1};
void dfs(vector<string>& b, int r, int c,
         Trie* nd, vector<bool>& vis,
         vector<string>& res){
  char ch=b[r][c];
  if(!nd->ch[ch]) return;  // prune!
  nd=nd->ch[ch];
  if(!nd->word.empty()) res.push_back(nd->word);
  int R=b.size(),C=b[0].size();
  vis[r*C+c]=true;
  for(int d=0;d<8;d++){
    int nr=r+dx[d],nc=c+dy[d];
    if(nr>=0&&nr<R&&nc>=0&&nc<C&&!vis[nr*C+nc])
      dfs(b,nr,nc,nd,vis,res);
  }
  vis[r*C+c]=false;  // backtrack
}
vector<string> boggle(vector<string>& b,
    vector<string>& words){
  Trie* root=new Trie;
  for(auto& w:words) ins(root,w);
  int R=b.size(),C=b[0].size();
  vector<bool> vis(R*C,false);
  vector<string> res;
  for(int r=0;r<R;r++)
    for(int c=0;c<C;c++)
      dfs(b,r,c,root,vis,res);
  return res;
}
