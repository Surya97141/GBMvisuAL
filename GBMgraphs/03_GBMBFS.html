JavaScript — BFS on Tree Node
class Node {
  constructor(name){
   this.name=name;
   this.children=[]; }
  breadthFirstSearch(array){
   const queue=[this];
   while(queue.length){
    const node=queue.shift();
    array.push(node.name);
    for(const c of node.children)
     queue.push(c);
   }
   return array; }
}

#pragma GCC optimize("O3")
#include<bits/stdc++.h>
using namespace std;

struct Node {
  string name;
  vector<Node*> children;
  Node(string n): name(n) {}
};

// BFS — O(V) time, O(W) space (W=max width)
vector<string> bfs(Node* root){
  vector<string> res;
  queue<Node*> q;
  q.push(root);
  while(!q.empty()){
    Node* cur=q.front(); q.pop();  // dequeue front
    res.push_back(cur->name);      // visit
    for(Node* child : cur->children)
      q.push(child);               // enqueue children
  }
  return res;
}

// BFS level-by-level (e.g. for level-order output)
vector<vector<string>> bfsLevels(Node* root){
  vector<vector<string>> res;
  queue<Node*> q;
  q.push(root);
  while(!q.empty()){
    int sz=q.size();  // nodes at current level
    res.push_back({});
    while(sz--){
      Node* cur=q.front(); q.pop();
      res.back().push_back(cur->name);
      for(Node* c:cur->children) q.push(c);
    }
  }
  return res;
}
