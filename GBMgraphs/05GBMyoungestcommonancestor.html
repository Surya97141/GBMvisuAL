def getYoungestCommonAncestor(
    topAncestor, descOne, descTwo):
  def getDepth(node, top):
   depth=0
   while node!=top:
    depth+=1; node=node.ancestor
   return depth
  d1=getDepth(descOne,topAncestor)
  d2=getDepth(descTwo,topAncestor)
  # bring deeper node to same level
  while d1!=d2:
   if d1>d2:
    descOne=descOne.ancestor; d1-=1
   else:
    descTwo=descTwo.ancestor; d2-=1
  # climb together until they meet
  while descOne!=descTwo:
   descOne=descOne.ancestor
   descTwo=descTwo.ancestor
  return descOne

Node* lca(Node* top, Node* a, Node* b){
  int da=getDepth(a,top), db=getDepth(b,top);
  // bring deeper to same level
  while(da!=db){
    if(da>db){a=a->ancestor;da--;}
    else    {b=b->ancestor;db--;}
  }
  // climb together
  while(a!=b){
    a=a->ancestor;
    b=b->ancestor;
  }
  return a;          // LCA found
}
