int quickselect(
    vector<int>& a, int lo,int hi,int k){
  int pivot=a[hi], p=lo;
  for(int i=lo; i<hi; i++)
    if(a[i]<=pivot) swap(a[i],a[p++]);
  swap(a[p],a[hi]); // place pivot
  if(p==k) return a[p]; // found!
  if(k<p) return quickselect(a,lo,p-1,k);
  return quickselect(a,p+1,hi,k);
}

int qs(int* a,int lo,int hi,int k){
  int m=(lo+hi)>>1;
  if(a[lo]>a[m])swap(a[lo],a[m]);
  if(a[lo]>a[hi])swap(a[lo],a[hi]);
  if(a[m]>a[hi])swap(a[m],a[hi]);
  // a[hi] is now median-of-3
  int p=lo;
  for(int i=lo;i<hi;i++)
    if(a[i]<=a[hi])swap(a[i],a[p++]);
  swap(a[p],a[hi]);
  if(p==k)return a[p];
  return k<p?qs(a,lo,p-1,k):qs(a,p+1,hi,k);
}
