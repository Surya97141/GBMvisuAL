bool feasible(v,k,int maxDur){
  int stations=1, cur=0;
  for(int d:v){
    if(cur+d>maxDur){stations++;cur=0;}
    cur+=d;
  }
  return stations<=k;
}
int assembly(vector<int>& v,int k){
  int lo=*max_element(v),
      hi=accumulate(v,0), ans=hi;
  while(lo<=hi){
    int mid=lo+(hi-lo)/2;
    if(feasible(v,k,mid)){ans=mid;hi=mid-1;}
    else               lo=mid+1;
  }
  return ans;
}

#pragma GCC optimize("O3,unroll-loops")
#pragma GCC target("avx2,bmi,bmi2")
#include <bits/stdc++.h>
using namespace std;
void solve(){
  int n,k; cin>>n>>k;
  vector<int> v(n); for(auto&x:v)cin>>x;
  int lo=*max_element(v.begin(),v.end());
  long long hi=accumulate(v.begin(),v.end(),0LL);
  long long ans=hi;
  while(lo<=hi){
    long long mid=(lo+hi)>>1;
    // inline feasibility — avoids fn call overhead
    int st=1,cur=0;
    for(int d:v){if(cur+d>mid){st++;cur=0;}cur+=d;}
    if(st<=k){ans=mid;hi=mid-1;}
    else lo=mid+1;
  }
  cout<<ans<<'\n';
}
int main(){ios::sync_with_stdio(0);cin.tie(0);solve();}
