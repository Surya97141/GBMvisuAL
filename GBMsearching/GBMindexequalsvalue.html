C++ — Index Equals Value
int indexEqualsValue(
    vector<int>& a) {
  int lo=0,hi=a.size()-1,res=-1;
  while(lo<=hi){
    int mid=lo+(hi-lo)/2;
    int gap=a[mid]-mid;
    if(gap==0){ res=mid; hi=mid-1; }
 else if(gap<0)  lo=mid+1; // go right
 else             hi=mid-1; // go left
  }
  return res;
}

#pragma GCC optimize("O3,unroll-loops")
#pragma GCC target("avx2,bmi,bmi2")
#include <bits/stdc++.h>
using namespace std;
void solve(){
  int n; cin>>n;
  vector<int> a(n); for(auto&x:a)cin>>x;
  int lo=0,hi=n-1,res=-1;
  while(lo<=hi){
    int mid=(lo+hi)>>1; // bitshift /2
    int g=a[mid]-mid;
    if(!g){res=mid;hi=mid-1;} // !g ≡ g==0
    else (g<0?lo:hi)=(g<0?mid+1:mid-1);
  }
  cout<<res<<'\n';
}
int main(){ios::sync_with_stdio(0);cin.tie(0);solve();}
