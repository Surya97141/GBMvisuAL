C++ — Shifted Binary Search
int shiftedBS(vector<int>& a, int t) {
  int lo=0, hi=a.size()-1;
  while(lo <= hi) {
    int mid=lo+(hi-lo)/2;
    if(a[mid]==t) return mid;
    if(a[lo] <= a[mid]) { // left sorted
      if(a[lo]<=t && t<a[mid]) hi=mid-1;
      else                     lo=mid+1;
    } else {                   // right sorted
      if(a[mid]<t && t<=a[hi]) lo=mid+1;
      else                      hi=mid-1;
    }
  }
  return -1;
}

#pragma GCC optimize("O3,unroll-loops")
#pragma GCC target("avx2,bmi,bmi2")
#include <bits/stdc++.h>
using namespace std;
int shiftedBS(int* a,int n,int t){
  int lo=0,hi=n-1;
  while(lo<=hi){
    int mid=(lo+hi)>>1; // bitshift /2
    if(a[mid]==t)return mid;
    // left half sorted?
    bool ls=a[lo]<=a[mid];
    bool inL=ls&&a[lo]<=t&&t<a[mid];
    bool inR=!ls&&a[mid]<t&&t<=a[hi];
    if(inL) hi=mid-1;
    else if(inR) lo=mid+1;
    else if(ls) lo=mid+1;
    else hi=mid-1;
  }
  return -1;
}
