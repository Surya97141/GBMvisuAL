C++ — Median of Two Sorted Arrays
double median(vector<int>& A,
           vector<int>& B) {
  if(A.size()>B.size()) return median(B,A);
  int m=A.size(),n=B.size();
  int lo=0,hi=m,half=(m+n+1)/2;
  while(lo<=hi){
    int i=lo+(hi-lo)/2, j=half-i;
    int ml1=i?A[i-1]:INT_MIN;
    int mr1=i<m?A[i]:INT_MAX;
    int ml2=j?B[j-1]:INT_MIN;
    int mr2=j<n?B[j]:INT_MAX;
    if(ml1<=mr2 && ml2<=mr1){
      if((m+n)%2) return max(ml1,ml2);
      return(max(ml1,ml2)+min(mr1,mr2))/2.0;
    } else if(ml1>mr2) hi=i-1; // i too big
    else             lo=i+1; // i too small
  }
  return -1;
}

#pragma GCC optimize("O3,unroll-loops")
#include <bits/stdc++.h>
using namespace std;
double med(int* A,int m,int* B,int n){
  if(m>n)return med(B,n,A,m);
  int lo=0,hi=m,h=(m+n+1)/2;
  while(lo<=hi){
    int i=(lo+hi)>>1,j=h-i;
    int l1=i?A[i-1]:INT_MIN;
    int r1=i<m?A[i]:INT_MAX;
    int l2=j?B[j-1]:INT_MIN;
    int r2=j<n?B[j]:INT_MAX;
    if(l1<=r2&&l2<=r1){
      if((m+n)&1)return max(l1,l2); // &1 = odd
      return(max(l1,l2)+min(r1,r2))*0.5;
    }
    l1>r2?hi=i-1:lo=i+1;
  }
  return -1;
}
