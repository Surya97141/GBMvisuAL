C++ — Binary Search
int binarySearch(
    vector<int>& arr, int target) {
  int lo=0, hi=arr.size()-1;
  while (lo <= hi) {
    int mid = lo+(hi-lo)/2;
    if (arr[mid]==target) return mid;
    else if (arr[mid]>target) hi=mid-1;
    else                    lo=mid+1;
  }
  return -1; // not found
}

#pragma GCC optimize("O3,unroll-loops")
#pragma GCC target("avx2")
#include <bits/stdc++.h>
using namespace std;
int binarySearch(
    int* arr, int n, int target) {
  int lo=0, hi=n-1;
  while (lo < hi) { // branchless-friendly
    int mid=(lo+hi)>>1; // >> instead of /2
    if (arr[mid] < target) lo=mid+1;
    else               hi=mid;
  }
  return (arr[lo]==target)?lo:-1;
}

int bsSTL(int* a, int n, int t){
  auto it=lower_bound(a,a+n,t);
  return(it!=a+n&&*it==t)?it-a:-1;
}
