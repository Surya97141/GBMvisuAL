C++ — Find Three Largest Numbers
vector<int> findThreeLargest(
    vector<int>& arr) {
  vector<int> top(3, INT_MIN);
  for (int n : arr) {
    if (n > top[2]) {
      top[0]=top[1]; top[1]=top[2];
      top[2] = n; // new first
    } else if (n > top[1]) {
      top[0] = top[1];
      top[1] = n; // new second
    } else if (n > top[0]) {
      top[0] = n; // new third
    }
  }
  return top;
}
