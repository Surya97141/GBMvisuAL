vector<int> searchRange(
    vector<int>& a, int t) {
  return {bound(a,t,true),bound(a,t,false)};
}
int bound(vector<int>& a,int t,bool left){
  int lo=0,hi=a.size()-1,res=-1;
  while(lo<=hi){
    int mid=lo+(hi-lo)/2;
    if(a[mid]==t){
      res=mid;
      if(left) hi=mid-1; // keep going left
      else     lo=mid+1; // keep going right
    } else if(a[mid]<t) lo=mid+1;
    else               hi=mid-1;
  }
  return res;
}
