C++ — Search In Sorted Matrix
vector<int> searchMatrix(
    vector<vector<int>>& m, int t) {
  int r=0, c=m[0].size()-1;
  while(r<m.size() && c>=0) {
    int cur = m[r][c];
    if(cur==t) return {r,c};
    else if(cur<t) r++; // go down
    else          c--; // go left
  }
  return {-1,-1};
}

}
