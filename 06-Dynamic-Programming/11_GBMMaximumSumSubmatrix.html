int maxSumSubmatrix(vector<vector<int>>& m,
                    int s) {
  int R=m.size(), C=m[0].size();
  // (R+1)×(C+1) prefix table, zero-padded
  vector<vector<int>> p(R+1,vector<int>(C+1,0));
  for(int i=1;i<=R;i++)
   for(int j=1;j<=C;j++)
    p[i][j] = m[i-1][j-1]   // ↙ cell
            + p[i-1][j]      // ↑ above
            + p[i][j-1]      // ← left
            - p[i-1][j-1];  // ↖ corner (−)
  int best = INT_MIN;
  for(int r1=0; r1<=R-s; r1++) {
   int r2 = r1+s-1;
   for(int c1=0; c1<=C-s; c1++) {
    int c2 = c1+s-1;
    int sum = p[r2+1][c2+1]  // ↘ full rect
            - p[r1][c2+1]   // − top strip
            - p[r2+1][c1]   // − left strip
            + p[r1][c1];     // + corner back
    if(sum > best) best = sum;
   }}
  return best;
}

#pragma GCC optimize("O3,unroll-loops") // compiler: max speed + unroll
#pragma GCC target("avx2")              // enable SIMD 256-bit instructions
#include <bits/stdc++.h>               // every std header in one line
using namespace std;
typedef long long ll;                   // ll saves typing, avoids overflow
typedef vector<vector<int>> vvi;        // shorthand for 2D vector

int solve(vvi& m, int s){
  int R=m.size(),C=m[0].size();
  vvi p(R+1,vector<int>(C+1,0));        // (R+1)×(C+1), zero border
  for(int i=1;i<=R;i++)
    for(int j=1;j<=C;j++)
      p[i][j]=m[i-1][j-1]+p[i-1][j]+p[i][j-1]-p[i-1][j-1];
  int best=INT_MIN;                      // -2^31, safe for any negative sum
  for(int r1=0;r1<=R-s;r1++){
    int r2=r1+s-1;
    for(int c1=0;c1<=C-s;c1++){
      int c2=c1+s-1;
      best=max(best,p[r2+1][c2+1]    // max() avoids if-branch = faster
        -p[r1][c2+1]-p[r2+1][c1]+p[r1][c1]);
    }}
  return best;
}

int main(){
  ios::sync_with_stdio(false);          // decouple C and C++ I/O streams
  cin.tie(nullptr);                    // stop cin flushing cout before read
  int R,C,s;
  cin>>R>>C>>s;
  vvi m(R,vector<int>(C));
  for(auto& row:m)
    for(auto& x:row) cin>>x;
  cout<<solve(m,s)<<'\n';              // '\n' faster than endl (no flush)
}
