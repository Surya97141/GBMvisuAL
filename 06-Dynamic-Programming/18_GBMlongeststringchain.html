vector<string> longestChain(
  vector<string>& words){
  sort(words.begin(),words.end(),
    [](auto& a,auto& b){return a.size()<b.size();});
  unordered_map<string,int> dp;
  unordered_map<string,string> prev;
  string best="";
  for(auto& w:words){
   dp[w]=1;
   for(int i=0;i<w.size();i++){
    string s=w.substr(0,i)+w.substr(i+1);
    if(dp.count(s)&&dp[s]+1>dp[w]){
     dp[w]=dp[s]+1; prev[w]=s;}}
   if(dp[w]>dp[best]) best=w;}
  // backtrack
  vector<string> res;
  for(string c=best;!c.empty();c=prev[c])
   res.push_back(c);
  return dp[best]>1?res:vector<string>{}; }

#pragma GCC optimize("O3,unroll-loops")
#pragma GCC target("avx2")
#include<bits/stdc++.h>
using namespace std;

vector<string> solve(vector<string>& w){
  sort(w.begin(),w.end(),            // sort by length ascending
    [](auto& a,auto& b){return a.size()<b.size();});
  unordered_map<string,int> dp;
  unordered_map<string,string> pre; // predecessor for backtrack
  string best;
  for(auto& s:w){
    dp[s]=1;
    for(int i=0;i<(int)s.size();i++){
      string t=s.substr(0,i)+s.substr(i+1); // delete char i
      if(dp.count(t)&&dp[t]+1>dp[s])
        dp[s]=dp[t]+1,pre[s]=t;         // shorter word exists!
    }
    if(dp[s]>dp[best]) best=s;
  }
  vector<string> res;
  for(string c=best;!c.empty();c=pre[c])  // follow predecessor chain
    res.push_back(c);
  return dp[best]>1?res:vector<string>{};
}

int main(){
  ios::sync_with_stdio(false); cin.tie(nullptr);
  int n; cin>>n;
  vector<string> w(n);
  for(auto& s:w) cin>>s;
  for(auto& s:solve(w)) cout<<s<<'\n';
}
