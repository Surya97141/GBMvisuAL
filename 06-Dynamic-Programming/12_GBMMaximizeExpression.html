int maximizeExpr(vector<int>& a) {
  int n=a.size(); if(n<4) return 0;
  // dp1[i] = best a[a] for a in [0..i]
  vector<int> dp1(n); dp1[0]=a[0];
  for(int i=1;i<n;i++)
    dp1[i]=max(dp1[i-1],a[i]);
  // dp2[i] = best dp1[b-1] - a[b]
  vector<int> dp2(n); dp2[1]=dp1[0]-a[1];
  for(int i=2;i<n;i++)
    dp2[i]=max(dp2[i-1],dp1[i-1]-a[i]);
  // dp3[i] = best dp2[c-1] + a[c]
  vector<int> dp3(n); dp3[2]=dp2[1]+a[2];
  for(int i=3;i<n;i++)
    dp3[i]=max(dp3[i-1],dp2[i-1]+a[i]);
  // dp4[i] = best dp3[d-1] - a[d]
  vector<int> dp4(n); dp4[3]=dp3[2]-a[3];
  for(int i=4;i<n;i++)
    dp4[i]=max(dp4[i-1],dp3[i-1]-a[i]);
  return dp4[n-1]; }

#pragma GCC optimize("O3,unroll-loops")  // max compiler speed
#pragma GCC target("avx2")               // SIMD 256-bit
#include<bits/stdc++.h>                // all headers
using namespace std;

int solve(vector<int>& a){
  int n=a.size();
  if(n<4) return 0;
  vector<int> d1(n),d2(n),d3(n),d4(n);
  d1[0]=a[0];
  for(int i=1;i<n;i++) d1[i]=max(d1[i-1],a[i]);
  d2[1]=d1[0]-a[1];                        // b starts at 1
  for(int i=2;i<n;i++) d2[i]=max(d2[i-1],d1[i-1]-a[i]);
  d3[2]=d2[1]+a[2];                        // c starts at 2
  for(int i=3;i<n;i++) d3[i]=max(d3[i-1],d2[i-1]+a[i]);
  d4[3]=d3[2]-a[3];                        // d starts at 3
  for(int i=4;i<n;i++) d4[i]=max(d4[i-1],d3[i-1]-a[i]);
  return d4[n-1];                          // answer = last cell of dp4
}

int main(){
  ios::sync_with_stdio(false);             // decouple C/C++ I/O
  cin.tie(nullptr);                     // stop cin flushing cout
  int n; cin>>n;
  vector<int> a(n);
  for(auto& x:a) cin>>x;
  cout<<solve(a)<<'\n';               // '\n' faster than endl
}
