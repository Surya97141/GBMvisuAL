int diceThrows(int numDice, int sides,
             int target) {
  // dp[d][s]=ways to get sum s with d dice
  vector<vector<long>> dp(
    numDice+1, vector<long>(target+1,0));
  dp[0][0]=1; // 0 dice → 1 way to sum 0
  for(int d=1; d<=numDice; d++)
   for(int s=1; s<=target; s++)
    for(int f=1; f<=sides&&f<=s; f++)
     dp[d][s]+=dp[d-1][s-f]; // add face f
  return dp[numDice][target]; }

#pragma GCC optimize("O3,unroll-loops")
#pragma GCC target("avx2")
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int solve(int d,int s,int t){
  vector<vector<ll>> dp(d+1,vector<ll>(t+1,0));
  dp[0][0]=1;                            // base: 0 dice, sum 0 = 1 way
  for(int i=1;i<=d;i++)
    for(int j=1;j<=t;j++)
      for(int f=1;f<=s&&f<=j;f++)
        dp[i][j]+=dp[i-1][j-f];          // face f contributes dp[i-1][j-f]
  return dp[d][t];                      // answer at bottom-right
}

int main(){
  ios::sync_with_stdio(false);

  int d,s,t; cin>>d>>s>>t;
  cout<<solve(d,s,t)<<'\n';
}
