int maxProfit(vector<int>& p, int k){
  int n=p.size();
  // dp[t][d] = max profit, t txns, day d
  vvi dp(k+1,vector<int>(n,0));
  for(int t=1;t<=k;t++){
   for(int d=1;d<n;d++){
    dp[t][d]=dp[t][d-1]; // no txn today
    for(int b=0;b<d;b++){
     int c=p[d]-p[b]+dp[t-1][b];
     dp[t][d]=max(dp[t][d],c);}}
  } return dp[k][n-1]; }

#pragma GCC optimize("O3,unroll-loops")
#pragma GCC target("avx2")
#include<bits/stdc++.h>
using namespace std;
typedef vector<vector<int>> vvi;

int solve(vector<int>& p,int k){
  int n=p.size();
  vvi dp(k+1,vector<int>(n,0));
  for(int t=1;t<=k;t++)
    for(int d=1;d<n;d++){
      dp[t][d]=dp[t][d-1];          // skip: no new txn on day d
      for(int b=0;b<d;b++)
        dp[t][d]=max(dp[t][d],      // sell on d, buy on b
          p[d]-p[b]+dp[t-1][b]);   // + best (t-1) txns before b
    }
  return dp[k][n-1];
}

int main(){
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  int n,k; cin>>n>>k;
  vector<int> p(n);
  for(auto& x:p) cin>>x;
  cout<<solve(p,k)<<'\n';
}
