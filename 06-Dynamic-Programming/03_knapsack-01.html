pair<int,vector<int>> knapsack(
  vector<vector<int>>& items, int cap){
  int n=items.size();
  // (n+1)×(cap+1) table, all zeros
  vvi dp(n+1,vector<int>(cap+1,0));
  for(int i=1;i<=n;i++){
   int v=items[i-1][0], w=items[i-1][1];
   for(int c=0;c<=cap;c++){
    dp[i][c]=dp[i-1][c];           // skip
    if(c>=w)                        // can take?
     dp[i][c]=max(dp[i][c],          // take
       v+dp[i-1][c-w]);}
  }
  // Backtrack to find chosen items
  vector<int> chosen; int c=cap;
  for(int i=n;i>0;i--)
   if(dp[i][c]!=dp[i-1][c]){
    chosen.push_back(i-1);
    c-=items[i-1][1]; }
  return make_pair(dp[n][cap],chosen);}

using namespace std;
typedef vector<vector<int>> vvi;

pair<int,vector<int>> knapsack(vvi& it,int cap){
  int n=it.size();
  vvi dp(n+1,vector<int>(cap+1,0));   // (n+1)×(cap+1) table
  for(int i=1;i<=n;i++){
    int v=it[i-1][0],w=it[i-1][1];
    for(int c=0;c<=cap;c++){
      dp[i][c]=dp[i-1][c];               // always try skip
      if(c>=w) dp[i][c]=max(dp[i][c],v+dp[i-1][c-w]);
    }}                                   // take if fits → max
  vector<int> chosen; int c=cap;
  for(int i=n;i>0;i--)               // backtrack top→bottom
    if(dp[i][c]!=dp[i-1][c])           // value changed → item taken
      {chosen.push_back(i-1);c-=it[i-1][1];}
  return make_pair(dp[n][cap],chosen);
}

int main(){
  ios::sync_with_stdio(false);          // decouple C/C++ I/O
  cin.tie(nullptr);                    // stop cin flushing cout
  int n,cap; cin>>n>>cap;
  vvi it(n,vector<int>(2));
  for(auto& x:it) cin>>x[0]>>x[1];
  auto [val,idx]=knapsack(it,cap);    // C++17 structured binding
  cout<<val<<'\n';                    // '\n' faster than endl
  for(int i:idx) cout<<i<<' ';
}
