vector<int> juiceBottling(
    vector<int>& prices) {
  int n=prices.size();
  // dp[i]=max rev, split[i]=best bottle
  vector<int> dp(n,0), split(n,0);
  for(int i=1;i<n;i++){
   for(int j=1;j<=i;j++){
    int cand=prices[j]+dp[i-j];
    if(cand>dp[i]){
      dp[i]=cand; split[i]=j;}}
  }
  // Backtrack split[] to get bottles
  vector<int> bottles;
  for(int rem=n-1;rem>0;){
   bottles.push_back(split[rem]);
   rem-=split[rem]; }
  sort(bottles.begin(),bottles.end());
  return bottles; }

vector<int> solve(vector<int>& p){
  int n=p.size();
  vector<int> dp(n,0),sp(n,0);  // dp=max rev, sp=split
  for(int i=1;i<n;i++)
    for(int j=1;j<=i;j++)
      if(p[j]+dp[i-j]>dp[i])    // bottle j + best(i-j)
        dp[i]=p[j]+dp[i-j], sp[i]=j;
  vector<int> res;
  for(int r=n-1;r>0;)          // backtrack split chain
    res.push_back(sp[r]), r-=sp[r];
  sort(res.begin(),res.end());    // ascending order required
  return res;
}

int main(){
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  int n; cin>>n;
  vector<int> p(n);
  for(auto& x:p) cin>>x;
  auto r=solve(p);
  for(int x:r) cout<<x<<' ';
}
