bool squareOfZeroes(vvi& m){
  int n=m.size();
  // p[i][j]=zeros in col j, rows 0..i
  vvi p(n+1,vector<int>(n,0));
  for(int i=0;i<n;i++)
   for(int j=0;j<n;j++)
    p[i+1][j]=p[i][j]+(1-m[i][j]);
  // try all squares (r1,c1)→(r2,c2)
  for(int r1=0;r1<n;r1++)
   for(int c1=0;c1<n;c1++)
    for(int r2=r1+1;r2<n;r2++){
     int c2=c1+(r2-r1);
     if(c2>=n) break;
     // check 4 borders using p[]
     int top=zerosInRow(m,r1,c1,c2);
     int bot=zerosInRow(m,r2,c1,c2);
     int lft=p[r2+1][c1]-p[r1][c1];
     int rgt=p[r2+1][c2]-p[r1][c2];
     int sz=r2-r1+1;
     if(top==sz&&bot==sz&&lft==sz&&rgt==sz)
      return true; }
  return false; }

#include<bits/stdc++.h>
using namespace std;
typedef vector<vector<int>> vvi;

bool solve(vvi& m){
  int n=m.size();
  // col prefix: p[i][j] = zeros in col j rows 0..i-1
  vvi p(n+1,vector<int>(n,0));
  for(int i=0;i<n;i++)
    for(int j=0;j<n;j++)
      p[i+1][j]=p[i][j]+(1-m[i][j]);  // 1→0 zero, 0→1 zero
  auto rowZ=[&](int r,int c1,int c2){  // zeros in row r, cols c1..c2
    int cnt=0;
    for(int c=c1;c<=c2;c++) cnt+=(1-m[r][c]);
    return cnt;};
  for(int r1=0;r1<n;r1++)
    for(int c1=0;c1<n;c1++)
      for(int r2=r1+1;r2<n;r2++){
        int c2=c1+(r2-r1);
        if(c2>=n) break;
        int sz=r2-r1+1;
        int top=rowZ(r1,c1,c2), bot=rowZ(r2,c1,c2);
        int lft=p[r2+1][c1]-p[r1][c1];    // O(1) col query
        int rgt=p[r2+1][c2]-p[r1][c2];
        if(top==sz&&bot==sz&&lft==sz&&rgt==sz) return true;
      }
  return false;
}

int main(){
  ios::sync_with_stdio(false); cin.tie(nullptr);
  int n; cin>>n;
  vvi m(n,vector<int>(n));
  for(auto& r:m) for(auto& x:r) cin>>x;
  cout<<(solve(m)?"true":"false")<<'\n';
}
