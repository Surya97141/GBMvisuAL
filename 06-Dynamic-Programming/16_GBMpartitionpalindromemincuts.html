int palindromeMinCuts(string s){
  int n=s.size();
  // Phase 1: build isPalin[i][j]
  vvb P(n,vector<bool>(n,false));
  for(int i=n-1;i>=0;i--)
   for(int j=i;j<n;j++)
    P[i][j]=(s[i]==s[j])&&
     (j-i<3||P[i+1][j-1]);
  // Phase 2: min cuts dp
  vector<int> cuts(n,0);
  for(int i=1;i<n;i++){
   if(P[0][i]){cuts[i]=0;continue;}
   cuts[i]=cuts[i-1]+1; // worst case
   for(int j=1;j<=i;j++)
    if(P[j][i])
     cuts[i]=min(cuts[i],cuts[j-1]+1);
  } return cuts[n-1]; }

using namespace std;

int solve(string& s){
  int n=s.size();
  vector<vector<bool>> P(n,vector<bool>(n,false));
  for(int i=n-1;i>=0;i--)        // fill bottom-up
    for(int j=i;j<n;j++)
      P[i][j]=(s[i]==s[j])&&(j-i<3||P[i+1][j-1]);
      // s[i..j] palindrome iff ends match AND inner is palindrome
  vector<int> cuts(n,0);
  for(int i=1;i<n;i++){
    if(P[0][i]){cuts[i]=0;continue;}  // whole prefix is palindrome
    cuts[i]=cuts[i-1]+1;               // worst: cut just before i
    for(int j=1;j<=i;j++)
      if(P[j][i]) cuts[i]=min(cuts[i],cuts[j-1]+1);
      // if s[j..i] is palindrome, cut at j → 1 + best before j
  }
  return cuts[n-1];
}

int main(){
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  string s; cin>>s;
  cout<<solve(s)<<'\n';
}
