int findPeak(vector<int>& a) {
  int L=0, R=n-1;
  while (L < R) {
    int mid = L+(R-L)/2;
    if (a[mid] < a[mid+1])
      L = mid+1; // upslope → peak is right
    else R = mid; // downslope → peak is here or left
  } return L; // L==R, peak found
}
