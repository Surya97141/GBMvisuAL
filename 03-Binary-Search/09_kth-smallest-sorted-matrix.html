📋 Problem Statement
Given an n×n matrix where each row and column is sorted in ascending order, find the Kth smallest element overall.

Key constraint: rows sorted left→right, columns sorted top→bottom. The matrix is not globally sorted — you cannot flatten and binary search directly.

int kthSmallest(
    vector<vector<int>>& m, int k) {
  int n=m.size();
  int lo=m[0][0], hi=m[n-1][n-1];
  while (lo < hi) {
    int mid = lo+(hi-lo)/2;
    int cnt = countLE(m, mid);
    if (cnt < k)
      lo = mid + 1; // too few ≤ mid
    else hi = mid;  // enough ≤ mid
  } return lo;
}
int countLE(m, mid) { // O(n)
  int cnt=0, r=0, c=n-1;
  while (r < n) {
    while(c>=0 && m[r][c]>mid) c--;
    cnt += c+1; r++;
  } return cnt;
}
