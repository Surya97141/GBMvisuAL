int findMin(vector<int>& a) {
  int lo=0, hi=a.size()-1;
  int ans = a[0];   // start: first elem
  while (lo <= hi) {
    if (a[lo] <= a[hi]) { // subarray sorted
      ans = min(ans, a[lo]);
      break;          // a[lo] is min here
    } int mid = lo+(hi-lo)/2;
    ans = min(ans, a[mid]);
    if (a[lo] <= a[mid]) { // left sorted
      lo = mid + 1;  // min in right half
    } else {             // right sorted
      hi = mid - 1;  // min in left half
    }
  } return ans;
