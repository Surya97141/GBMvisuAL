bool searchMatrix(vector<vector<int>>& m,
                   int target) {
  int rows=m.size(), cols=m[0].size();
  int lo=0, hi=rows*cols-1;
  while (lo <= hi) {
    int mid = lo+(hi-lo)/2;
    int val = m[mid/cols][mid%cols];
    if (val == target)
      return true;   // found!
    else if (val < target)
      lo = mid + 1;  // go right
    else
      hi = mid - 1;  // go left
  }
  return false;    // not found
}
