int binarySearch(vector<int>& a, int t){
  int L=0, R=n-1;
  while (L <= R) {
    int mid = L + (R-L)/2;
    if (a[mid] == t) return mid;
    else if (a[mid] < t)
      L = mid + 1; // target in right half
    else
      R = mid - 1; // target in left half
  }
  return -1; // not found
}
