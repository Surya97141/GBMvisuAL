// Can we achieve ≤ capacity with ≤ days?
bool feasible(int cap, vector<int>& w, int d){
  int days=1, cur=0;
  for(auto x:w){
    if(cur+x>cap){days++;cur=0;}
    cur+=x; }
  return days<=d; }
int solve(vector<int>& w, int d) {
  int L=max(w), R=sum(w), ans=R;
  while(L<=R){
    int mid=L+(R-L)/2;
    if(feasible(mid,w,d))
      {ans=mid; R=mid-1;} // try smaller
    else L=mid+1; // need more capacity
  } return ans; }
