// Lower Bound: first i where a[i] >= t
int lowerBound(vector<int>& a, int t) {
  int L=0, R=n, ans=n;
  while(L<R){ int m=L+(R-L)/2;
    if(a[m]>=t){ans=m; R=m;}  // go left
    else L=m+1;              // go right
  } return ans; }
// Upper Bound: first i where a[i] > t
int upperBound(vector<int>& a, int t) {
  int L=0, R=n, ans=n;
  while(L<R){ int m=L+(R-L)/2;
    if(a[m]>t){ans=m; R=m;}   // go left
    else L=m+1;              // go right
  } return ans; }
