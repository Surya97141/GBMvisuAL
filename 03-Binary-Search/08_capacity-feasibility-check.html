📋 Problem Statement
You have — packages on a conveyor belt with weights —. A ship makes one trip per day, loading packages in order (no reordering allowed) up to its capacity limit per trip.

Find the minimum ship capacity so that all packages are delivered within D days.

Key constraints: capacity ≥ max(weights) (heaviest package must fit), capacity ≤ sum(weights) (worst case: ship everything in one day).

int shipWithinDays(
    vector<int>& w, int D) {
  int lo=*max_element(w.begin(),w.end());
  int hi=accumulate(w.begin(),w.end(),0);
  int ans=hi;  // worst case: all in 1 day
  while (lo <= hi) {
    int mid = lo+(hi-lo)/2;
    if (canShip(w, mid, D)) {
      ans = mid;  // feasible → record
      hi = mid-1; // try smaller capacity
    } else lo=mid+1; // need more capacity
  } return ans;
}
bool canShip(w, cap, D) {
  int days=1, load=0;
  for (int x : w) {
    load += x;
    if (load > cap)
      { days++; load=x; }
  }
  return days <= D;
}
