C++ — Generate Div Tags
void bt(int openLeft, int closeLeft,
     string& cur, vector<string>& res) {
  if (closeLeft == 0) {
    res.push_back(cur); return;
  }
  if (openLeft > 0) {
    cur += "<div>";
    bt(openLeft-1, closeLeft, cur, res);
    cur.erase(cur.size()-5); // backtrack
  }
  if (closeLeft > openLeft) {
    cur += "</div>";
    bt(openLeft, closeLeft-1, cur, res);
    cur.erase(cur.size()-6); // backtrack
  }
}
