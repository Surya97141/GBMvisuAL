#include<bits/stdc++.h>
using namespace std;
bool isValid(auto& b,int r,int c,int d){
  for(int i=0;i<9;i++){
   if(b[r][i]==d) return false; // row
   if(b[i][c]==d) return false; // col
   int br=3*(r/3)+i/3, bc=3*(c/3)+i%3;
   if(b[br][bc]==d) return false; // box
  } return true; }
bool solve(vector<vector<int>>& b){
  for(int r=0;r<9;r++)
   for(int c=0;c<9;c++){
    if(b[r][c]!=0) continue;
    for(int d=1;d<=9;d++){
     if(!isValid(b,r,c,d)) continue;
     b[r][c]=d;
     if(solve(b)) return true;
     b[r][c]=0; // backtrack
    }
    return false; // no digit works
   }
  return true; // all filled!
}

// O(1) constraint check using bitsets
bitset<10> rowUsed[9],colUsed[9],boxUsed[9];
auto box=[&](int r,int c){return(r/3)*3+c/3;};

bool canPlace(int r,int c,int d){
  return !rowUsed[r][d]&&!colUsed[c][d]
        &&!boxUsed[box(r,c)][d];
}
// MRV: pick cell with fewest valid digits
// → reduces backtracking dramatically
pair<int,int> pickMRV(){
  int best=10,br=-1,bc=-1;
  for(int r=0;r<9;r++)
    for(int c=0;c<9;c++)
      if(b[r][c]==0){
        int cnt=0;
        for(int d=1;d<=9;d++)
          if(canPlace(r,c,d)) cnt++;
        if(cnt<best){best=cnt;br=r;bc=c;}
      }
  return{br,bc};
}
