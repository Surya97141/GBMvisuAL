C++ — Non-Attacking Queens
void bt(int row, int n,
    set cols, set diag, set anti,
    vector<int>& queens, int& count) {
  if (row == n) { count++; return; }
  for (int col=0; col<n; col++) {
    if (cols.count(col) ||
        diag.count(row-col) ||
        anti.count(row+col)) continue;
    cols.insert(col); queens[row]=col;
    diag.insert(row-col);
    anti.insert(row+col);
    bt(row+1,n,cols,diag,anti,queens,count);
    cols.erase(col); diag.erase(row-col);
    anti.erase(row+col); // backtrack
  }
}
