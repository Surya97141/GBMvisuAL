C++ — Product Sum
int productSum(
    vector<any>& arr, int depth=1) {
  int sum = 0;
  for (auto& elem : arr) {
    if (holds_alternative<vector<any>>(elem))
      sum += productSum(
          get<vector<any>>(elem), depth+1);
    else
      sum += get<int>(elem);
  }
  return depth * sum;
}
