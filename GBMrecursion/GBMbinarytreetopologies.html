C++ — Binary Tree Topologies
int numTopologies(int n) {
  vector<int> dp(n+1, 0);
  dp[0] = 1; // null tree = 1 topology
  for (int k=1; k<=n; k++) {
    for (int i=0; i<k; i++) {
      // left=i nodes, right=k-1-i nodes
      dp[k] += dp[i] * dp[k-1-i];
    }
  }
  return dp[n];
}
