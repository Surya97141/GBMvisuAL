C++ — Ambiguous Measurements
bool canMeasure(cups, lo, hi,
    low, high, memo) {
  if (lo>=low && hi<=high) return true;
  if (lo>high) return false; // too high
  if (memo.count({lo,hi})) return memo[{lo,hi}];
  bool res = false;
  for (auto [L,H] : cups) {
    if (canMeasure(cups,
        lo+L, hi+H, low, high, memo)) {
      res = true; break;
    }
  }
  return memo[{lo,hi}] = res;
}
