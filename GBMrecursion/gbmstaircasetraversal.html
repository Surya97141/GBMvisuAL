C++ — Sliding Window DP
int staircaseTraversal(
    int height, int maxSteps) {
  vector<int> dp(height+1, 0);
  dp[0] = 1; // 1 way to stand at base
  int windowSum = 0;
  for (int i=1; i<=height; i++) {
    windowSum += dp[i-1];   // add new entry
    if (i > maxSteps)         // remove expired
      windowSum -= dp[i-maxSteps-1];
    dp[i] = windowSum;
  }
  return dp[height];
}
