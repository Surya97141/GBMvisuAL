C++ — Blackjack Probability
float blackjackProb(
    int target, int hand,
    unordered_map<int,float>& memo) {
  if (hand > target) return 1.0; // bust
  if (hand >= target-4) return 0.0; // stand
  if (memo.count(hand)) return memo[hand];
  float p = 0.0;
  for (int card=1; card<=10; card++)
    p += 0.1 * blackjackProb(
           target, hand+card, memo);
  return memo[hand] = p;
}
