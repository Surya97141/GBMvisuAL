C++ — Interweaving Strings
bool interweaving(
    string one, string two, string three,
    int i, int j, map<pair<int,int>,bool>& m){
  if(i+j==three.size()) return true;
  if(m.count({i,j})) return m[{i,j}];
  bool res = false;
  if(i<one.size()&&one[i]==three[i+j])
    res = interweaving(one,two,three,i+1,j,m);
  if(!res&&j<two.size()&&two[j]==three[i+j])
    res = interweaving(one,two,three,i,j+1,m);
  return m[{i,j}] = res;
}
