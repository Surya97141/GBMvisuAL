C++ — Reveal Minesweeper
void reveal(board, int r, int c) {
  if (board[r][c]=="M") {
    board[r][c]="X"; return; // lose
  }
  int mines = countAdj(board,r,c);
  if (mines > 0) {
    board[r][c]=to_string(mines); return;
  }
  board[r][c]="0"; // safe zone flood
  for (auto [nr,nc] : neighbors(r,c))
    if (board[nr][nc]=="H")
      reveal(board, nr, nc);
}
