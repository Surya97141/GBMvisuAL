void helper(vector<int>& arr,
    vector<int>& cur, vector<bool>& used,
    vector<vector<int>>& res) {
  if (cur.size()==arr.size()) {
    res.push_back(cur); return;
  }
  for (int i=0; i<arr.size(); i++) {
    if (used[i]) continue;
    used[i]=true;
    cur.push_back(arr[i]);
    helper(arr,cur,used,res);
    cur.pop_back();         // backtrack
    used[i]=false;         // unmark
  }
}
 
vector<vector<int>> permutations(
    vector<int> arr) {
  vector<vector<int>> res;
  vector<int> cur; vector<bool> used(arr.size(),false);
  helper(arr,cur,used,res);
  return res;
}
