C++ — Lowest Common Manager
struct Result { OrgChart* lcm; int cnt; };
Result helper(OrgChart* node,
    OrgChart* r1, OrgChart* r2) {
  int cnt = 0;
  for (auto child : node->directReports) {
    auto res = helper(child, r1, r2);
    if (res.lcm) return res; // found!
    cnt += res.cnt;
  }
  if (node==r1 || node==r2) cnt++;
  if (cnt == 2) return {node, cnt};
  return {nullptr, cnt};
}
