C++ — Phone Number Mnemonics
const vector<string> PAD = {
  "0","1","abc","def","ghi",
  "jkl","mno","pqrs","tuv","wxyz"};
void bt(string& phone, int idx,
     string& cur, vector<string>& res) {
  if (idx==phone.size()){
    res.push_back(cur); return;
  }
  for (char c : PAD[phone[idx]-'0']) {
    cur.push_back(c);
    bt(phone, idx+1, cur, res);
    cur.pop_back(); // backtrack
  }
}
