import heapq
def aStarAlgorithm(sr,sc,er,ec,graph):
  def h(r,c): # Manhattan distance
   return abs(r-er)+abs(c-ec)
  g={}; came={}
  g[(sr,sc)]=0; f0=h(sr,sc)
  heap=[(f0,sr,sc)]
  openSet={(sr,sc)}
  closed=set()
  while heap:
   _,r,c=heapq.heappop(heap)
   if (r,c) in closed: continue
   closed.add((r,c))
   if r==er and c==ec:
    return reconstruct(came,r,c)

#pragma GCC optimize("O3")
#include<bits/stdc++.h>
using namespace std;
typedef tuple<int,int,int> tiii;

vector<pair<int,int>> aStar(
    vector<vector<int>>& g,
    int sr,int sc,int er,int ec){
  int R=g.size(),C=g[0].size();
  auto h=[&](int r,int c){
    return abs(r-er)+abs(c-ec);};  // Manhattan
  vector<vector<int>> gScore(R,vector<int>(C,INT_MAX));
  vector<vector<pair<int,int>>> came(R,vector<pair<int,int>>(C,{-1,-1}));
  priority_queue<tiii,vector<tiii>,greater<tiii>> pq;
  gScore[sr][sc]=0;
  pq.push({h(sr,sc),sr,sc});
  int dx[]={-1,1,0,0},dy[]={0,0,-1,1};
  while(!pq.empty()){
    auto [f,r,c]=pq.top(); pq.pop();
    if(f>gScore[r][c]+h(r,c)) continue; // stale
    if(r==er&&c==ec){
      vector<pair<int,int>> path;
      for(auto [pr,pc]=make_pair(r,c);
          pr!=-1;tie(pr,pc)=came[pr][pc])
        path.push_back({pr,pc});
      reverse(path.begin(),path.end());
      return path;
    }
    for(int d=0;d<4;d++){
      int nr=r+dx[d],nc=c+dy[d];
      if(nr<0||nr>=R||nc<0||nc>=C||g[nr][nc]) continue;
      int ng=gScore[r][c]+1;
      if(ng<gScore[nr][nc]){
        gScore[nr][nc]=ng; came[nr][nc]={r,c};
        pq.push({ng+h(nr,nc),nr,nc});
      }
    }
  }
  return {};
}
