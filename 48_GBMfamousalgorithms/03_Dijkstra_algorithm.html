C++ — Dijkstra Implementation
vector<int> dijkstra(
  int start, vvii& edges){
  int n=edges.size();
  vector<int> dist(n,INT_MAX); // ∞
  dist[start]=0;
  priority_queue<pii,
   vector<pii>,greater<pii>> pq;
  pq.push({0,start});
  while(!pq.empty()){
   auto [d,u]=pq.top(); pq.pop();
   if(d>dist[u]) continue; // stale
   for(auto [v,w]:edges[u]){
    int nd=dist[u]+w;
    if(nd<dist[v]){
     dist[v]=nd; pq.push({nd,v});}}
  }
  for(auto& x:dist)
   if(x==INT_MAX) x=-1;
  return dist; }

#pragma GCC optimize("O3,unroll-loops")
#pragma GCC target("avx2")
#include<bits/stdc++.h>
using namespace std;
typedef pair<int,int> pii;
typedef vector<vector<pii>> vvii;

vector<int> dijkstra(int s, vvii& g){
  int n=g.size();
  vector<int> d(n,INT_MAX);
  d[s]=0;
  priority_queue<pii,vector<pii>,greater<pii>> pq;
  pq.push({0,s});                       // {dist, node} — dist first for ordering
  while(!pq.empty()){
    auto [dist,u]=pq.top(); pq.pop();  // structured binding (C++17)
    if(dist>d[u]) continue;            // stale entry — skip
    for(auto [v,w]:g[u])
      if(d[u]+w<d[v])
        d[v]=d[u]+w, pq.push({d[v],v});
  }
  for(auto& x:d) if(x==INT_MAX) x=-1; // unreachable = -1
  return d;
}

int main(){
  ios::sync_with_stdio(false); cin.tie(nullptr);
  int n,s; cin>>n>>s;
  vvii g(n);
  int u,v,w;
  while(cin>>u>>v>>w) g[u].push_back({v,w});
  for(int x:dijkstra(s,g)) cout<<x<<' ';
}
