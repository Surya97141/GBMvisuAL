def kruskals(edges):
  n=len(edges)
  parent=list(range(n)); rank=[0]*n
  def find(x):
   if parent[x]!=x: parent[x]=find(parent[x])
   return parent[x]
  # gather all unique edges
  allEdges=[]
  for i,nbrs in enumerate(edges):
   for nb,w in nbrs:
    if i<nb: allEdges.append((w,i,nb))
  allEdges.sort() # sort by weight
  mst=[[] for _ in range(n)]
  for w,u,v in allEdges:
   ru,rv=find(u),find(v)
   if ru==rv: continue # cycle!
   if rank[ru]<rank[rv]: ru,rv=rv,ru
   parent[rv]=ru
   if rank[ru]==rank[rv]: rank[ru]+=1
   mst[u].append([v,w]); mst[v].append([u,w])
  return mst

#pragma GCC optimize("O3,unroll-loops")
#include<bits/stdc++.h>
using namespace std;

struct DSU{
  vector<int> p,r;
  DSU(int n):p(n),r(n,0){iota(p.begin(),p.end(),0);}
  int find(int x){return p[x]==x?x:p[x]=find(p[x]);}
  bool unite(int a,int b){
    a=find(a);b=find(b);
    if(a==b)return false;       // same component → cycle
    if(r[a]<r[b])swap(a,b);
    p[b]=a; if(r[a]==r[b])r[a]++;
    return true;
  }
};

long long kruskal(int n,
    vector<tuple<int,int,int>> edges){
  sort(edges.begin(),edges.end()); // sort by weight
  DSU dsu(n);
  long long total=0; int cnt=0;
  for(auto [w,u,v]:edges){
    if(dsu.unite(u,v)){  // different components
      total+=w; cnt++;
      if(cnt==n-1)break;  // MST complete
    }
  }
  return total;             // MST weight
}

int main(){
  ios::sync_with_stdio(false); cin.tie(nullptr);
  int n,m; cin>>n>>m;
  vector<tuple<int,int,int>> edges(m);
  for(auto& [w,u,v]:edges) cin>>u>>v>>w;
  cout<<kruskal(n,edges)<<'\n';
}
