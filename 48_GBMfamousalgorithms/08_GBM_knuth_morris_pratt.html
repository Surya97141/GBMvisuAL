def kmpSearch(string, sub):
  n,m=len(string),len(sub)
  # Phase 1: build lps (failure function)
  lps=[0]*m; i=1; j=0
  while i<m:
   if sub[i]==sub[j]:
    lps[i]=j+1; i+=1; j+=1
   elif j>0: j=lps[j-1]
   else: lps[i]=0; i+=1
  # Phase 2: search
  i=0; j=0
  while i<n:
   if string[i]==sub[j]:
    i+=1; j+=1
   if j==m: return True
   elif i<n and string[i]!=sub[j]:
    if j>0: j=lps[j-1]
    else: i+=1
  return False

#pragma GCC optimize("O3,unroll-loops")
#include<bits/stdc++.h>
using namespace std;

// Returns all start indices of pat in txt
vector<int> kmp(string txt, string pat){
  int n=txt.size(), m=pat.size();
  vector<int> lps(m,0), res;
  // Build LPS
  for(int i=1,j=0;i<m;){
    if(pat[i]==pat[j]) lps[i++]=++j;
    else if(j) j=lps[j-1];
    else lps[i++]=0;
  }
  // Search
  for(int i=0,j=0;i<n;){
    if(txt[i]==pat[j]){i++;j++;}
    if(j==m){
      res.push_back(i-m);  // found at index i-m
      j=lps[j-1];           // continue for more occurrences
    } else if(i<n&&txt[i]!=pat[j]){
      if(j) j=lps[j-1];    // smart jump, no backtrack on i
      else i++;
    }
  }
  return res;                // empty = not found
}

int main(){
  ios::sync_with_stdio(false); cin.tie(nullptr);
  string txt,pat; cin>>txt>>pat;
  auto v=kmp(txt,pat);
  cout<<(!v.empty()?"true":"false")<<'\n';
}
