def stableInternships(interns,teams):
  n=len(interns)
  # rank[t][intern]=preference rank of intern for team t
  rank={t:{intern:i for i,intern
    in enumerate(teams[t])} for t in range(n)}
  free=[True]*n; nxt=[0]*n
  matched={} # team→intern
  while any(free):
   intern=free.index(True)
   team=interns[intern][nxt[intern]]
   nxt[intern]+=1
   if team not in matched:
    matched[team]=intern; free[intern]=False
   else:
    cur=matched[team]
    if rank[team][intern]<rank[team][cur]:
     matched[team]=intern; free[intern]=False
     free[cur]=True  # old intern freed
  return [[i,t] for t,i in matched.items()]

vector<pair<int,int>> stableMatch(
    vector<vector<int>>& interns,
    vector<vector<int>>& teams){
  int n=interns.size();
  // rank[t][intern] = preference of intern for team t
  vector<vector<int>> rank(n,vector<int>(n));
  for(int t=0;t<n;t++)
    for(int k=0;k<n;k++)
      rank[t][teams[t][k]]=k;          // lower rank = more preferred
  vector<int> nxt(n,0),matched(n,-1),who(n,-1);
  queue<int> free;
  for(int i=0;i<n;i++) free.push(i);
  while(!free.empty()){
    int intern=free.front(); free.pop();
    int team=interns[intern][nxt[intern]++];
    if(who[team]==-1){             // team is free → accept
      who[team]=intern; matched[intern]=team;
    } else {
      int cur=who[team];
      if(rank[team][intern]<rank[team][cur]){ // team prefers new intern
        who[team]=intern; matched[intern]=team;
        matched[cur]=-1; free.push(cur); // old intern freed
      } else free.push(intern);            // team rejects new intern
    }
  }
  vector<pair<int,int>> res;
  for(int i=0;i<n;i++) res.push_back({i,matched[i]});
  return res;
}
