Python — Union Find with Path Compression + Union by Rank
class UnionFind:
  def __init__(self):
    self.parents={}
    self.ranks={}

  def createSet(self,v):
    # each value starts as its own parent
    self.parents[v]=v
    self.ranks[v]=0  # rank=0: no children

  def find(self,v):
    if v not in self.parents:
      return None
    if self.parents[v]!=v:
      # path compression: point to root
      self.parents[v]=self.find(
        self.parents[v])
    return self.parents[v]

  def union(self,v1,v2):
    r1=self.find(v1)
    r2=self.find(v2)
    if r1 is None or r2 is None:
      return
    if r1==r2: return  # same set
    # union by rank: attach smaller to larger
    if self.ranks[r1]<self.ranks[r2]:
      self.parents[r1]=r2
    elif self.ranks[r1]>self.ranks[r2]:
      self.parents[r2]=r1
    else:
      self.parents[r2]=r1
      self.ranks[r1]+=1

#pragma GCC optimize("O3,unroll-loops")
#include<bits/stdc++.h>
using namespace std;

struct DSU {
  vector<int> par,rnk;
  DSU(int n):par(n),rnk(n,0){
    iota(par.begin(),par.end(),0);  // par[i]=i
  }
  int find(int x){
    if(par[x]!=x)
      par[x]=find(par[x]);        // path compression
    return par[x];
  }
  bool unite(int a,int b){
    a=find(a); b=find(b);
    if(a==b) return false;    // same set
    if(rnk[a]<rnk[b]) swap(a,b); // attach b under a
    par[b]=a;
    if(rnk[a]==rnk[b]) rnk[a]++;
    return true;
  }
  bool same(int a,int b){
    return find(a)==find(b);
  }
};

int main(){
  ios::sync_with_stdio(false); cin.tie(nullptr);
  int n; cin>>n;
  DSU dsu(n);
  // unite edges, check connectivity, etc.
}
