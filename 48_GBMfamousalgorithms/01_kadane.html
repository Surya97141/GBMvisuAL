vector<int> kadane(vector<int>& a){
  int cur=a[0],best=a[0];
  int s=0,e=0,ts=0; // best/temp start
  for(int i=1;i<a.size();i++){
   if(cur+a[i]>a[i]){        // extend
    cur+=a[i];
   }else{                     // reset
    cur=a[i]; ts=i;}
   if(cur>best){              // new best?
    best=cur; s=ts; e=i;}}
  return {a.begin()+s,
          a.begin()+e+1}; }

// Returns max sum only (O(1) space)
int kadaneSum(vector<int>& a){
  int cur=a[0], best=a[0];
  for(int i=1;i<(int)a.size();i++){
    cur=max(a[i],cur+a[i]);       // extend or restart
    best=max(best,cur);
  }
  return best;
}

// Returns actual subarray [s..e]
vector<int> kadane(vector<int>& a){
  int cur=a[0],best=a[0],s=0,e=0,ts=0;
  for(int i=1;i<(int)a.size();i++){
    if(cur+a[i]>a[i]) cur+=a[i];   // extend current window
    else{cur=a[i];ts=i;}            // restart from i
    if(cur>best){best=cur;s=ts;e=i;}
  }
  return {a.begin()+s,a.begin()+e+1};
}
