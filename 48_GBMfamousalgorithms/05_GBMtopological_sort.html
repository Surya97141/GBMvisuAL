def topoSort(jobs, deps):
  graph={j:[] for j in jobs}
  inDeg={j:0 for j in jobs}
  for a,b in deps:
   graph[a].append(b); inDeg[b]+=1
  queue=[j for j in jobs if inDeg[j]==0]
  result=[]
  while queue:
   job=queue.pop(0)  # dequeue
   result.append(job)
   for nb in graph[job]:
    inDeg[nb]-=1
    if inDeg[nb]==0:
     queue.append(nb)
  if len(result)!=len(jobs):
   return []  # cycle detected
  return result

#pragma GCC optimize("O3,unroll-loops")
#include<bits/stdc++.h>
using namespace std;

vector<int> topoSort(int n,
    vector<pair<int,int>>& deps){
  vector<vector<int>> g(n);
  vector<int> indeg(n,0);
  for(auto [a,b]:deps)
    g[a].push_back(b), indeg[b]++;
  queue<int> q;
  for(int i=0;i<n;i++)
    if(!indeg[i]) q.push(i);    // no prerequisites
  vector<int> res;
  while(!q.empty()){
    int u=q.front(); q.pop();
    res.push_back(u);
    for(int v:g[u])
      if(--indeg[v]==0) q.push(v); // newly ready
  }
  if((int)res.size()!=n) return {}; // cycle!
  return res;
}

int main(){
  ios::sync_with_stdio(false); cin.tie(nullptr);
  int n,m; cin>>n>>m;
  vector<pair<int,int>> deps(m);
  for(auto& [a,b]:deps) cin>>a>>b;
  for(int x:topoSort(n,deps)) cout<<x<<' ';
}
