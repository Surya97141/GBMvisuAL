import heapq
def primsAlgorithm(edges):
  n=len(edges)
  visited=[False]*n
  mst=[[] for _ in range(n)]
  heap=[(w,0,v) for v,w in edges[0]]
  heapq.heapify(heap); visited[0]=True
  while heap:
   w,src,dst=heapq.heappop(heap)
   if visited[dst]: continue
   visited[dst]=True
   mst[src].append([dst,w])
   mst[dst].append([src,w])
   for nb,nw in edges[dst]:
    if not visited[nb]:
     heapq.heappush(heap,(nw,dst,nb))
  return mst

#pragma GCC optimize("O3,unroll-loops")
#include<bits/stdc++.h>
using namespace std;
typedef pair<int,int> pii;
typedef vector<vector<pii>> vvii;

long long prim(vvii& g){
  int n=g.size();
  vector<bool> vis(n,false);
  // {weight, vertex} min-heap
  priority_queue<pii,vector<pii>,greater<pii>> pq;
  pq.push({0,0});                // start from vertex 0
  long long total=0; int cnt=0;
  while(!pq.empty()&&cnt<n){
    auto [w,u]=pq.top(); pq.pop();
    if(vis[u]) continue;      // stale
    vis[u]=true; total+=w; cnt++;
    for(auto [v,wv]:g[u])
      if(!vis[v]) pq.push({wv,v}); // add frontier edges
  }
  return total;
}

int main(){
  ios::sync_with_stdio(false); cin.tie(nullptr);
  int n,m; cin>>n>>m;
  vvii g(n);
  int u,v,w;
  while(cin>>u>>v>>w)
    g[u].push_back({v,w}), g[v].push_back({u,w});
  cout<<prim(g)<<'\n';
}
