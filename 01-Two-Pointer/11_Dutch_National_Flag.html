void sortColors(vector<int>& a) {
  int low=0, mid=0, high=n-1;
  while (mid <= high) {
    if (a[mid] == 0) {
      swap(a[low], a[mid]);
      low++; mid++;
    } else if (a[mid] == 1) {
      mid++;     // already in place
    } else {    // a[mid] == 2
      swap(a[mid], a[high]);
      high--;    // mid stays! unknown came back
    }
  }
} // sorted in-place, O(n) time O(1) space
