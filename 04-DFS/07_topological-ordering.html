📋 Problem Statement
Given a Directed Acyclic Graph (DAG), produce a topological ordering of its nodes — a linear sequence where for every directed edge u → v, node u appears before node v.

DFS approach: Run DFS. When a node finishes (all descendants explored), prepend it to the result. The final list is a valid topological order. If a cycle is detected (GRAY neighbor seen), no valid order exists.

vector<int> topoSort(int n, adj) {
  vector<int> color(n,0); // 0=W 1=G 2=B
  vector<int> result;
  for(int i=0; i<n; i++)
    if(color[i]==WHITE)
      dfs(i, color, result);
  return result; // already in topo order
}
void dfs(int u, color, result) {
  color[u] = GRAY;  // on active path
  for(int v : adj[u]) {
    if(color[v]==WHITE)
      dfs(v, color, result);
    else if(color[v]==GRAY)
      throw "cycle!"; // not a DAG
  }
  color[u] = BLACK;
  result.insert(result.begin(), u); // prepend
}
