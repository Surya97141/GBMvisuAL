C++ — Flood Fill DFS
void floodFill(grid, r, c,
           oldC, newC) {
  if(r<0||r>=rows) return;
  if(c<0||c>=cols) return;
  if(grid[r][c]!=oldC) return;
  grid[r][c] = newC; // paint!
  floodFill(grid,r-1,c,oldC,newC);
  floodFill(grid,r+1,c,oldC,newC);
  floodFill(grid,r,c-1,oldC,newC);
  floodFill(grid,r,c+1,oldC,newC);
}
