C++ — All 3 Traversals
void preorder(node* n) {
  if(!n) return;
  visit(n); // ROOT first
  preorder(n->left);
  preorder(n->right); }
void inorder(node* n) {
  if(!n) return;
  inorder(n->left);
  visit(n); // ROOT middle
  inorder(n->right); }
void postorder(node* n) {
  if(!n) return;
  postorder(n->left);
  postorder(n->right);
  visit(n); // ROOT last
}
