vector<int> findAnagrams(string s, string p){
  map<char,int> need, have;
  for(char c:p) need[c]++;
  int formed=0, req=need.size();
  for(int R=0; R<n; R++){
    have[s[R]]++;
    if(need.count(s[R])&&
      have[s[R]]==need[s[R]]) formed++;
    if(R-L+1 > p.size()){
      if(need.count(s[L])&&
        have[s[L]]==need[s[L]]) formed--;
      have[s[L]]--; L++;
    }
    if(formed==req) res.push_back(L);
  } return res; }
