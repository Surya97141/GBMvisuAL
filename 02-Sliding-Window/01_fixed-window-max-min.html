int fixedWindowMax(vector<int>& a, int k) {
  int L=0, winSum=0, ans=INT_MIN;
  for (int R=0; R<n; R++) {
    winSum += a[R];   // expand right
    if (R >= k-1) {  // window full
      ans = max(ans, winSum);
      winSum -= a[L]; L++; // slide
    }
  } return ans; }
