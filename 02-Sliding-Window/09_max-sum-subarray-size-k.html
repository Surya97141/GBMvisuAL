int maxSumK(vector<int>& a, int k) {
  int L=0, winSum=0, ans=INT_MIN;
  for (int R=0; R<n; R++) {
    winSum += a[R];  // expand right
    if (R-L+1 == k) {  // window full
      ans = max(ans, winSum);
      winSum -= a[L++]; // slide left
    }
  } return ans; }
