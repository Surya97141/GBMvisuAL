int countAtMostK(string& s, int k){
  map<char,int> freq; int L=0,cnt=0;
  for(int R=0; R<n; R++){
    freq[s[R]]++;
    while(freq.size() > k){
      if(--freq[s[L]]==0)
        freq.erase(s[L]);
      L++;
    }
    cnt += R-L+1; // all subarrays ending at R
  } return cnt; }
