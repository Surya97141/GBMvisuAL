int minSubLen(vector<int>& a, int x) {
  int L=0, sum=0, ans=INT_MAX;
  for (int R=0; R<n; R++) {
    sum += a[R];   // expand right
    while (sum >= x) { // valid — try shrink
      ans = min(ans, R-L+1);
      sum -= a[L++]; // shrink from left
    }
  } return ans==INT_MAX ? -1 : ans; }
