double bestAvg(vector<int>& p, int k) {
  double sum=0, best=-1e9;
  int L=0;
  for (int R=0; R<n; R++) {
    sum += p[R];  // add right element
    if (R-L+1 == k) { // window full
      best = max(best, sum/k);
      sum -= p[L++]; // slide: remove left
    }
  } return best; }
