string minWindow(string s, string t) {
  unordered_map<char,int> need, have;
  for (char c : t) need[c]++;
  int required = need.size();
  int formed = 0, L = 0;
  int bestLen = INT_MAX, bestL = 0;
  for (int R = 0; R < s.size(); R++) {
    have[s[R]]++;
    if (need.count(s[R]) &&
        have[s[R]] == need[s[R]])
      formed++;
    while (formed == required) {
      if (R-L+1 < bestLen) {
        bestLen=R-L+1; bestL=L; }
      have[s[L]]--;
      if (need.count(s[L]) &&
          have[s[L]] < need[s[L]])
        formed--;
      L++;
    } // end while
  } // end for
  return bestLen==INT_MAX
    ? "" : s.substr(bestL, bestLen);
}
