Node* middleNode(Node* head) {
  Node *slow=head, *fast=head;
  while (fast != nullptr &&
         fast->next != nullptr) {
    slow = slow->next;       // +1
    fast = fast->next->next; // +2
  }
  return slow; // middle!
}
