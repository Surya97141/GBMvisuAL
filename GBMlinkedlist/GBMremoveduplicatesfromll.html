C++ — Remove Duplicates From Linked List
Node* removeDups(Node* head) {
  Node* cur = head;
  while (cur != nullptr) {
    while (cur->next &&
           cur->next->val == cur->val)
      cur->next = cur->next->next; // skip
    cur = cur->next; // advance
  }
  return head;
}

#pragma GCC optimize("O3")
#include <bits/stdc++.h>
using namespace std;
struct Node{int v;Node*n;Node(int x):v(x),n(nullptr){};};
// pool allocator for competitive use
Node* remDups(Node* h){
  for(Node*c=h;c;c=c->n)
    while(c->n&&c->n->v==c->v)
      c->n=c->n->n; // no delete needed in CF
  return h;
}
// with delete (memory-safe version):
Node* remDupsSafe(Node* h){
  for(Node*c=h;c;c=c->n){
    Node*t=c->n;
    while(t&&t->v==c->v){Node*d=t;t=t->n;delete d;}
    c->n=t;
  }
  return h;
}
