C++ — Merging Linked Lists
LinkedList* mergingLinkedLists(
    LinkedList* l1, LinkedList* l2){
  // Phase 1: measure lengths
  int lenA=0, lenB=0;
  LinkedList* a=l1; LinkedList* b=l2;
  while(a){lenA++;a=a->next;}
  while(b){lenB++;b=b->next;}
  // Phase 2: advance longer list
  a=l1; b=l2;
  while(lenA>lenB){a=a->next;lenA--;}
  while(lenB>lenA){b=b->next;lenB--;}
  // Phase 3: advance together
  while(a!=b){
   a=a->next;
   b=b->next;
  }
  return a; // null if no intersection
}
⚡ Why length-equalization works
// If two lists intersect, they share a tail.
// Let:  m = L1 length,  n = L2 length
//       k = shared tail length
// L1 unique prefix = m - k nodes
// L2 unique prefix = n - k nodes
//
// If we skip abs(m-n) nodes from the longer
// list, both pointers are now equidistant
// from the intersection.
// Advancing together → they meet at it.
//
// If no intersection: both reach null
// simultaneously (a == b == null) → return null
//
// Alternative O(1) space trick (no lengths):
//   When a reaches null, redirect to head of L2.
//   When b reaches null, redirect to head of L1.
//   After at most m+n steps, a==b (intersection or null).
//   Both travel exactly m+n steps total!
