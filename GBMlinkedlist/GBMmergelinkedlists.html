C++ — Merge Linked Lists
LinkedList* mergeLinkedLists(
    LinkedList* l1, LinkedList* l2){
  LinkedList *p1=l1, *p2=l2;
  LinkedList *p1Prev=nullptr;
  while(p1&&p2){
   if(p1->value<=p2->value){
    p1Prev=p1;
    p1=p1->next;
   }else{
    if(p1Prev) p1Prev->next=p2;
    else l1=p2; // new head
    p1Prev=p2;
    p2=p2->next;
    p1Prev->next=p1;
   }
  }
  // attach remaining tail
  if(p1Prev) p1Prev->next=p2;
  else l1=p2;
  return l1;
}
⚡ Key insight: weaving in-place
// p1 walks L1, p2 walks L2, p1Prev is the
// tail of the merged result so far.
//
// Case A: p1.val <= p2.val
//   p1 already in correct position.
//   Just advance p1Prev and p1.
//
// Case B: p2.val < p1.val
//   p2 node must be INSERTED before p1.
//   ① p1Prev.next = p2   (splice p2 in)
//   ② p2 = p2.next       (advance p2)
//   ③ p1Prev.next.next = p1  (p2 points to p1)
//   p1Prev slides to the inserted node.
//   p1 stays (still needs to be compared).
//
// When one list empties:
//   Remaining tail of the other is already
//   sorted — just attach it. O(1) work!
