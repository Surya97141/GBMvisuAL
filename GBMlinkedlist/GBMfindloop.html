C++ — Find Loop (Floyd's)
LinkedList* findLoop(LinkedList* head){
  LinkedList *slow=head->next;
  LinkedList *fast=head->next->next;
  // Phase 1: find meeting point
  while(slow!=fast){
   slow=slow->next;
   fast=fast->next->next;
  }
  // Phase 2: reset slow to head
  slow=head;
  while(slow!=fast){
   slow=slow->next;
   fast=fast->next;
  }
  return slow; // loop origin
}
⚡ Floyd's Math — Why Phase 2 works
// Let:  F = distance from head to loop start
//        C = loop circumference
//        D = distance from loop start to meeting pt
//
// At meeting point:
//   slow traveled: F + D
//   fast traveled: F + D + kC  (lapped k times)
//   fast = 2×slow:
//   F + D + kC = 2(F + D)
//   → kC = F + D
//   → F = kC - D
//
// So: distance from head to loop origin (F)
//     equals distance from meeting point
//     to loop origin (kC - D)!
//
// Reset slow to head, keep fast at meeting pt.
// Both advance by 1. They meet at loop origin!
