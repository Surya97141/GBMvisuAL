C++ — Sum of Linked Lists
LinkedList* sumOfLinkedLists(
    LinkedList* l1, LinkedList* l2){
  LinkedList dummy(0);
  LinkedList* cur=&dummy;
  int carry=0;
  while(l1||l2||carry){
   int d1=l1?l1->value:0;
   int d2=l2?l2->value:0;
   int sum=d1+d2+carry;
   int digit=sum%10;
   carry=sum/10;
   cur->next=new LinkedList(digit);
   cur=cur->next;
   if(l1) l1=l1->next;
   if(l2) l2=l2->next;
  }
  return dummy.next;
}
⚡ Key insight: LSB first = perfect
// Normally, addition works right-to-left
// in a number. Linked lists are traversed
// left-to-right. So storing digits LSB-first
// means we naturally traverse in the
// CORRECT addition order: units → tens → …
//
// Loop condition: l1 || l2 || carry
// This handles 3 cases elegantly:
//   l1 exhausted → d1 = 0
//   l2 exhausted → d2 = 0
//   Both done but carry=1 → emit extra node
//
// Dummy head trick: avoids special-casing
// the first result node. dummy.next is the
// real head of the result list.
//
// sum = d1 + d2 + carry_in
// digit = sum % 10   ← what goes in result
// carry_out = sum / 10  ← 0 or 1
