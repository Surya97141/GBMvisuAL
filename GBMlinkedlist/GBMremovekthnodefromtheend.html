C++ — Remove Kth Node From End
void removeKthNodeFromEnd(
    LinkedList* head, int k){
  int counter=1;
  LinkedList *first=head, *second=head;
  // Phase 1: advance first by k steps
  while(counter!=k){
   first=first->next;
   counter++;
  }
  // Phase 2: if first is at end, remove head
  if(first->next==nullptr){
   head->value=head->next->value;
   head->next=head->next->next;
   return;
  }
  // Phase 3: advance both until first reaches end
  while(first->next!=nullptr){
   first=first->next;
   second=second->next;
  }
  // second->next is the target
  second->next=second->next->next;
}
⚡ Why the k-gap works
// List has n nodes. Kth from end = (n-k)th from start.
//
// Phase 1: advance first by k-1 steps.
//   first is now at index k-1 (0-based).
//   Gap between first and second = k-1.
//
// Phase 2: advance both together.
//   When first reaches the LAST node (index n-1),
//   second has moved (n-1)-(k-1) = n-k steps.
//   second is now at index n-k-1 = one BEFORE the target.
//   → second->next is the kth from end!
//
// Special case: if after Phase 1, first->next==nullptr,
//   then k == n → remove the HEAD.
//   Since we can't change the head pointer itself,
//   copy next node's value into head and skip next.
