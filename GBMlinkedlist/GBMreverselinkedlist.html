C++ — Reverse Linked List
LinkedList* reverseLinkedList(
    LinkedList* head){
  LinkedList *prev=nullptr;
  LinkedList *curr=head;
  while(curr!=nullptr){
   // ① save next
   LinkedList* nxt=curr->next;
   // ② flip arrow
   curr->next=prev;
   // ③ advance prev
   prev=curr;
   // ④ advance curr
   curr=nxt;
  }
  return prev; // new head
}
⚡ Why the 4-step order matters
// The 4 steps MUST happen in this exact order:
//
// ① nxt = curr->next    SAVE next first!
//    Without this, step ② destroys the
//    reference to the rest of the list.
//
// ② curr->next = prev   FLIP the arrow
//    curr now points backward to prev.
//    The rest of the list is saved in nxt.
//
// ③ prev = curr         SLIDE prev forward
//    prev is the "reversed tail" — it
//    becomes the new head when we finish.
//
// ④ curr = nxt          SLIDE curr forward
//    Move to the next unprocessed node.
//
// When curr == nullptr:
//    prev is the new head of the reversed list.
