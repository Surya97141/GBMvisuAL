Unbind helper (defensive — used before every insert)
void unbind(Node* n){
  if(n==head) head=head->next;
  if(n==tail) tail=tail->prev;
  if(n->prev) n->prev->next=n->next;
  if(n->next) n->next->prev=n->prev;
  n->prev=n->next=nullptr; // isolate
}
setHead / setTail
void setHead(Node* n){
  if(!head){head=tail=n;return;}
  insertBefore(head,n);
}
void setTail(Node* n){
  if(!tail){head=tail=n;return;}
  insertAfter(tail,n);
}
insertBefore / insertAfter
void insertBefore(Node* ref,Node* n){
  unbind(n);
  n->prev=ref->prev; n->next=ref;
  if(ref->prev) ref->prev->next=n;
  else head=n;
  ref->prev=n;
}
void insertAfter(Node* ref,Node* n){
  unbind(n);
  n->next=ref->next; n->prev=ref;
  if(ref->next) ref->next->prev=n;
  else tail=n;
  ref->next=n;
}
insertAtPosition / remove / contains
void insertAtPosition(int pos,Node* n){
  if(pos==1){setHead(n);return;}
  Node*c=head;
  for(int i=1;c&&i<pos-1;i++)c=c->next;
  c?insertAfter(c,n):setTail(n);
}
void remove(Node* n){unbind(n);}
void removeNodesWithValue(int v){
  Node*c=head;
  while(c){Node*nx=c->next;
    if(c->val==v)remove(c);c=nx;}
}
bool containsNodeWithValue(int v){
  Node*c=head;
  while(c){if(c->val==v)return true;c=c->next;}
  return false;}

#include<bits/stdc++.h>
using namespace std;
struct Node{int v;Node*p,*n;
  Node(int x):v(x),p(nullptr),n(nullptr){};
};
struct DLL{
  Node*h=nullptr,*t=nullptr;
  void ub(Node*x){
    if(x==h)h=h->n;
    if(x==t)t=t->p;
    if(x->p)x->p->n=x->n;
    if(x->n)x->n->p=x->p;
    x->p=x->n=nullptr;
  }
  void ib(Node*r,Node*x){
    ub(x);x->p=r->p;x->n=r;
    if(r->p)r->p->n=x;else h=x;
    r->p=x;
  }
  void ia(Node*r,Node*x){
    ub(x);x->n=r->n;x->p=r;
    if(r->n)r->n->p=x;else t=x;
    r->n=x;
  }
  void sh(Node*x){!h?h=t=x:ib(h,x);}
  void st(Node*x){!t?h=t=x:ia(t,x);}
  bool has(int v){
    for(auto*c=h;c;c=c->n)
      if(c->v==v)return true;
    return false;
  }
};
